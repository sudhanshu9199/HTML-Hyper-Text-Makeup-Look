    n = len(arr)
    
    for i in range(n):
        # Find the minimum element in the unsorted portion
        min_index = i
        for j in range(i+1, n):
            if arr[j] < arr[min_index]:
                min_index = j
                
        # Swap the found minimum element with the first element of the unsorted portion
